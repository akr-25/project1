## C++ code to write sum of my name in ASCII



```
#include <iostream>
using namespace std;

int main(){
	string name = "PRANAV AGARWAL";
	int sum = 0;

	for(int i = 0; i < name.length(); i++){
		sum += (int)name.at(i);
	}
	
	cout << sum;

	return 0;
}
```
